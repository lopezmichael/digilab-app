## Phase 1: Database Changes

All DB changes are additive (new columns, new table). No destructive changes. Can be applied to the live DB before any code changes since they don't affect existing behavior.

### 1A: Add `continent` column to `scenes`

**File:** `db/schema.sql` (add column definition)
**Migration:**
```sql
ALTER TABLE scenes ADD COLUMN IF NOT EXISTS continent TEXT;

-- Populate based on existing country values
UPDATE scenes SET continent = CASE
  WHEN country IN ('United States', 'Canada', 'Mexico') THEN 'north_america'
  WHEN country IN ('Brazil', 'Costa Rica', 'Colombia', 'Argentina', 'Chile', 'Peru') THEN 'south_america'
  WHEN country IN ('Germany', 'Italy', 'France', 'Spain', 'United Kingdom', 'Netherlands', 'Poland', 'Portugal', 'Sweden', 'Norway', 'Denmark', 'Finland', 'Austria', 'Switzerland', 'Belgium', 'Czech Republic', 'Romania', 'Hungary', 'Greece', 'Ireland') THEN 'europe'
  WHEN country IN ('Japan', 'South Korea', 'China', 'Taiwan', 'Philippines', 'Thailand', 'Malaysia', 'Singapore', 'Indonesia', 'India') THEN 'asia'
  WHEN country IN ('Australia', 'New Zealand') THEN 'oceania'
  WHEN country IN ('South Africa', 'Nigeria', 'Kenya', 'Egypt') THEN 'africa'
END
WHERE continent IS NULL AND scene_type = 'metro';

-- Online scenes get no continent (handled as special case)
-- Country-level parent scenes inherit from their metros
UPDATE scenes s SET continent = (
  SELECT DISTINCT continent FROM scenes child
  WHERE child.parent_scene_id = s.scene_id AND child.continent IS NOT NULL
  LIMIT 1
)
WHERE s.scene_type = 'country' AND s.continent IS NULL;
```

**Verification:** `SELECT scene_id, display_name, country, continent FROM scenes ORDER BY continent, country;`

### 1B: `admin_user_scenes` junction table + `regional_admin` role

**File:** `db/schema.sql` (add table definition)
**Migration:**
```sql
-- Create junction table
CREATE TABLE IF NOT EXISTS admin_user_scenes (
    user_id INTEGER NOT NULL REFERENCES admin_users(user_id) ON DELETE CASCADE,
    scene_id INTEGER NOT NULL REFERENCES scenes(scene_id) ON DELETE CASCADE,
    is_primary BOOLEAN DEFAULT FALSE,
    assigned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    assigned_by TEXT,
    PRIMARY KEY (user_id, scene_id)
);

CREATE INDEX IF NOT EXISTS idx_admin_user_scenes_scene ON admin_user_scenes(scene_id);
CREATE INDEX IF NOT EXISTS idx_admin_user_scenes_user ON admin_user_scenes(user_id);

-- Populate from existing 1:1 assignments
INSERT INTO admin_user_scenes (user_id, scene_id, is_primary, assigned_by)
SELECT user_id, scene_id, TRUE, 'migration'
FROM admin_users
WHERE scene_id IS NOT NULL AND is_active = TRUE
ON CONFLICT DO NOTHING;
```

**Role update:** The `role` column on `admin_users` currently supports `'super_admin'` and `'scene_admin'`. Add `'regional_admin'` as a valid value. No schema change needed (it's a TEXT column), but update any role-checking logic.

**Backwards compatibility:** Keep `scene_id` on `admin_users` for now. Code reads from junction table, writes to both. Remove column in a future release.

**Modified `get_scene_choices()` — add optgroup support:**
```r
get_scene_choices <- function(continent = "all") {
  # Query scenes filtered by continent
  if (continent == "all") {
    scenes <- safe_query(db_pool, "SELECT * FROM scenes WHERE is_active = TRUE AND scene_type IN ('metro', 'country') ORDER BY country, display_name")
  } else if (continent == "online") {
    return(list("Online / Webcam" = "online"))
  } else {
    scenes <- safe_query(db_pool, "SELECT * FROM scenes WHERE is_active = TRUE AND continent = $1 ORDER BY country, display_name", params = list(continent))
  }

  # Build optgroup structure: list of lists keyed by country
  choices <- list("All Scenes" = "all")

  countries <- unique(scenes$country)
  for (cty in countries) {
    country_scenes <- scenes[scenes$country == cty & scenes$scene_type == "metro", ]
    country_parent <- scenes[scenes$country == cty & scenes$scene_type == "country", ]

    if (nrow(country_scenes) == 0) next

    group_choices <- list()

    # Add "All of Country" if parent scene exists and there are 2+ metros
    if (nrow(country_parent) > 0 && nrow(country_scenes) >= 2) {
      group_choices[[paste0("All of ", cty)]] <- country_parent$slug[1]
    }

    # Add individual metros
    for (i in seq_len(nrow(country_scenes))) {
      group_choices[[country_scenes$display_name[i]]] <- country_scenes$slug[i]
    }

    choices[[cty]] <- group_choices
  }

  choices
}
```

**Save logic:**
```r
# On admin user save:
# 1. Upsert admin_users row (without scene_id changes)
# 2. Sync junction table:
conn <- pool::localCheckout(db_pool)
tryCatch({
  dbBegin(conn)

  # Clear existing assignments
  safe_execute(conn, "DELETE FROM admin_user_scenes WHERE user_id = $1",
               params = list(user_id))

  # Insert new assignments
  for (sid in input$admin_user_scenes) {
    safe_execute(conn, "INSERT INTO admin_user_scenes (user_id, scene_id, assigned_by) VALUES ($1, $2, $3)",
                 params = list(user_id, as.integer(sid), rv$admin_username))
  }

  # Also update legacy scene_id column (first assigned scene)
  safe_execute(conn, "UPDATE admin_users SET scene_id = $1 WHERE user_id = $2",
               params = list(as.integer(input$admin_user_scenes[1]), user_id))

  dbCommit(conn)
}, error = function(e) {
  dbRollback(conn)
  stop(e)
})
```
